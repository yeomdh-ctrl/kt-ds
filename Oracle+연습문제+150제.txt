-- SCALA QUERY
--  Select의 결과를 fetch 할 떄 마다 서브쿼리를 실행 => 실행된 단 하나의 ROW & COLUMN을 fetch 하는 방법
--  성능 하락의 주요 원인 이기에 사용 권장하지 않음 => 3000개를 fetch하면 Scala Query가 3000번 수행됨
--  사용을 해야할 때는 어쩔 수 없는 상황일 때
--  예) 어떤 데이터 하나를 조회하기 위해 하나 이상의 Join이 더 필요할 때
--      기존의 조인 테이블의 수 = 7개 => 추가 정보를 얻어오기 위해 1개의 조인을 더 필요로 할 때
--		=> 추가된 테이블이 기존의 조인과는 큰 관계가 없을 때 사용
-- 부서명과 도시명을 조회한다 - 조인사용 X
SELECT D.DEPARTMENT_NAME
	 , D.LOCATION_ID
	 , (SELECT CITY --SCALA 쿼리는 반드시 하나의 컬럼만 조회해야 한다 & 반드시 하나의 로우만 조회 되어야 한다.
	 	  FROM LOCATIONS L
	 	 WHERE L.LOCATION_ID = D.LOCATION_ID) AS CITY
	 , '도시의 이름은' || (SELECT CITY --SCALA 쿼리는 반드시 하나의 컬럼만 조회해야 한다 & 반드시 하나의 로우만 조회 되어야 한다.
	 	  				  FROM LOCATIONS L
	 					 WHERE L.LOCATION_ID = D.LOCATION_ID) || '입니다.'	 
  FROM DEPARTMENTS D
;
-- 도시명과 부서명을 조회한다
SELECT L.CITY
 	 , L.LOCATION_ID
 	 , (SELECT D.DEPARTMENT_NAME
 	 	  FROM DEPARTMENTS D
 	 	 WHERE D.LOCATION_ID = L.LOCATION_ID) AS DEPT_NAME
  FROM LOCATIONS L	 
;  

-- 글자 채우기 함수(LPAD(), RPAD())
-- LPAD(컬럼, 자릿수, 남은 자릿수 만큼 채울 문자) -> Left Padding(컬럼의 값 왼쪽에 자릿수만큼 채움)
-- RPAD(컬럼, 자릿수, 남은 자릿수 만큼 채울 문자) -> Right Padding(컬럼의 값 오른쪽에 자릿수만큼 채움)
-- 대소문자 바꾸기(UPPER(), LOWER())
-- UPPER(컬럼) --> 소문자들을 모두 대문자로 변경
-- LOWER(컬럼) --> 대문자들을 모두 소문자로 변경
-- 문자 연결 연산자(|| <- Concat)
-- 예) 'a' || 'b' => 'ab'
-- 문자열 바꾸기 함수(REPLACE())
-- 예) "Dong Hun" => "Dong_Hun": REPLACE('Dong Hun', ' ','_')
--    "Dong Hun" => "Dong Hun": REPLACE('Dong Hun', ' ')
WITH TEMP AS (SELECT 'Dong Hun' AS FIRST_NAME
	 			   , 'Yeom' AS LAST_NAME
	 			   , 'DONGHUNYEOM' AS EMAIL
	 			FROM DUAL)
SELECT FIRST_NAME
	 , LAST_NAME
	 , EMAIL
	 , LPAD(FIRST_NAME, 5, '-')
	 , RPAD(FIRST_NAME, 5, '-')
	 , LPAD(FIRST_NAME, 1, '_')
	 , UPPER(FIRST_NAME)
	 , LOWER(EMAIL)
	 , UPPER(FIRST_NAME || LAST_NAME)
	 , UPPER(REPLACE(FIRST_NAME, ' ') || LAST_NAME)
  FROM TEMP
;

 SELECT LEVEL

  FROM DUAL

	 , SYSDATE
  FROM DUAL
;

        INNER JOIN LOCATIONS L
           ON D.LOCATION_ID = L.LOCATION_ID

-- 130. 부서별 연봉 차이(최고연봉 - 최저연봉)가 가장 큰 부서명을 조회한다.
SELECT DEPARTMENT_NAME
  FROM(SELECT D.DEPARTMENT_NAME
       	    , MAX(E.SALARY) - MIN(E.SALARY) AS M_SAL
         FROM EMPLOYEES E
        INNER JOIN DEPARTMENTS D
           ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
        GROUP BY D.DEPARTMENT_NAME
        ORDER BY M_SAL DESC)
 WHERE ROWNUM = 1
;

SELECT JOB_TITLE
  FROM JOBS
 WHERE JOB_ID IN (SELECT DISTINCT JOB_ID
 				    FROM EMPLOYEES
 				   WHERE DEPARTMENT_ID IN (SELECT DISTINCT DEPARTMENT_ID
  									         FROM(SELECT D.DEPARTMENT_ID
       	   								 	           , MAX(E.SALARY) - MIN(E.SALARY) AS M_SAL
                                                    FROM EMPLOYEES E
                                                   INNER JOIN DEPARTMENTS D
                                                      ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
                                                   GROUP BY D.DEPARTMENT_ID
                                                   ORDER BY M_SAL DESC)
 										    WHERE ROWNUM = 1)) 
;

SELECT DEPARTMENT_NAME
	 , JOB_TITLE
  FROM(SELECT D.DEPARTMENT_NAME
       	    , J.JOB_TITLE
       	    , AVG(SALARY) AS AVG
         FROM EMPLOYEES E
        INNER JOIN DEPARTMENTS D
           ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
        INNER JOIN JOBS J
           ON E.JOB_ID = J.JOB_ID
        GROUP BY D.DEPARTMENT_NAME 
        	, J.JOB_TITLE
        ORDER BY AVG ASC)
 WHERE ROWNUM = 1       
;
