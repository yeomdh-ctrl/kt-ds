-- 1. 100번 사원의 모든 부하직원을 계층조회한다.
SELECT EMPLOYEE_ID
     , FIRST_NAME
     , LAST_NAME
     , EMAIL
     , PHONE_NUMBER
     , HIRE_DATE
     , JOB_ID
     , SALARY
     , COMMISSION_PCT
     , MANAGER_ID
     , DEPARTMENT_ID
   FROM EMPLOYEES
  START WITH EMPLOYEE_ID = 100
CONNECT BY PRIOR EMPLOYEE_ID = MANAGER_ID  
;
-- 2. 113번 사원의 모든 상사를 계층조회한다.
SELECT EMPLOYEE_ID
     , FIRST_NAME
     , LAST_NAME
     , EMAIL
     , PHONE_NUMBER
     , HIRE_DATE
     , JOB_ID
     , SALARY
     , COMMISSION_PCT
     , MANAGER_ID
     , DEPARTMENT_ID
   FROM EMPLOYEES
  START WITH EMPLOYEE_ID = 113
CONNECT BY PRIOR MANAGER_ID = EMPLOYEE_ID  
;
-- 3. IT 부서장의 모든 부하직원을 계층조회한다.
SELECT E.EMPLOYEE_ID
     , E.FIRST_NAME
     , E.LAST_NAME
     , E.EMAIL
     , E.PHONE_NUMBER
     , E.HIRE_DATE
     , E.JOB_ID
     , E.SALARY
     , E.COMMISSION_PCT
     , E.MANAGER_ID
     , E.DEPARTMENT_ID
   FROM EMPLOYEES E
  INNER JOIN DEPARTMENTS D
     ON E.DEPARTMENT_ID = D.DEPARTMENT_ID
  START WITH D.DEPARTMENT_NAME = 'IT'
CONNECT BY PRIOR E.EMPLOYEE_ID = E.MANAGER_ID 
;  
-- 4. 부서장들의 부하직원을 계층조회한다.
SELECT E.EMPLOYEE_ID
     , E.FIRST_NAME
     , E.LAST_NAME
     , E.EMAIL
     , E.PHONE_NUMBER
     , E.HIRE_DATE
     , E.JOB_ID
     , E.SALARY
     , E.COMMISSION_PCT
     , E.MANAGER_ID
     , E.DEPARTMENT_ID
   FROM EMPLOYEES E
  INNER JOIN EMPLOYEES M
     ON E.MANAGER_ID = M.EMPLOYEE_ID 
  START WITH E.MANAGER_ID IS NOT NULL
CONNECT BY PRIOR E.EMPLOYEE_ID = E.MANAGER_ID  
;
-- 5. 부서명이 가장 긴 부서에서 근무중인 사원의 모든 정보를 조회한다.
SELECT E.EMPLOYEE_ID
     , E.FIRST_NAME
     , E.LAST_NAME
     , E.EMAIL
     , E.PHONE_NUMBER
     , E.HIRE_DATE
     , E.JOB_ID
     , E.SALARY
     , E.COMMISSION_PCT
     , E.MANAGER_ID
     , E.DEPARTMENT_ID 
  FROM(SELECT LENGTH(DEPARTMENT_NAME) AS LEN_NAME
     		, DEPARTMENT_ID
 		 FROM DEPARTMENTS
 		ORDER BY LEN_NAME DESC) D
 INNER JOIN EMPLOYEES E
    ON D.DEPARTMENT_ID = E.DEPARTMENT_ID
 WHERE ROWNUM = 1   
; 
-- 6. 2002년부터 2006년까지 입사한 사원은 몇 명인지 연도별로 조회한다.
SELECT YEAR
     , COUNT(EMPLOYEE_ID)
  FROM (SELECT TO_CHAR(HIRE_DATE, 'YYYY') AS YEAR
		     , EMPLOYEE_ID
		  FROM EMPLOYEES)
  WHERE YEAR BETWEEN 2002 AND 2006
  GROUP BY YEAR
  ORDER BY YEAR ASC
; 
-- 7. 입사일이 가장 빠른 사원 5명의 이름과 입사일을 조회한다.
SELECT YEAR
 	 , FIRST_NAME
  FROM(SELECT YEAR
		    , FIRST_NAME
  		 FROM(SELECT TO_CHAR(HIRE_DATE, 'YYYY-MM-DD') AS YEAR
	       		   , FIRST_NAME 
 			    FROM EMPLOYEES)	 
 	    ORDER BY YEAR ASC)	 
 WHERE ROWNUM <= 5
; 
-- 8. 커미션을 받는 사원들의 이름과 커미션을 조회한다. 단, 커미션이 가장 높은 사원 3명은 제외한다.
SELECT COMMISSION_PCT
	 , FIRST_NAME
  FROM(SELECT COMMISSION_PCT
	 	    , FIRST_NAME
	 	    , ROWNUM AS R_NUM
  		 FROM EMPLOYEES
 		WHERE COMMISSION_PCT IS NOT NULL
 	    ORDER BY COMMISSION_PCT DESC)	 
 WHERE R_NUM >= 4 
;
